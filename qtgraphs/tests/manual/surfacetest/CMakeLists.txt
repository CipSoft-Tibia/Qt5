    GUI

        checkboxwrapper.cpp checkboxwrapper.h
        graphmodifier.cpp graphmodifier.h
        main.cpp
    )

    Qt::Gui

    PROPERTIES QT_RESOURCE_ALIAS "mapimage"
    )

    "Heightmap.png"
    "mapimage.png"
    )

    PREFIX
    "/maps"
    FILES
    ${surfacetest_resource_files}
    )
