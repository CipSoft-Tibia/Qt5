#####################################################################
## InputSupportPrivate Module:
#####################################################################

qt_internal_add_module(InputSupportPrivate
    CONFIG_MODULE_NAME input_support
    STATIC
    INTERNAL_MODULE
    DEFINES
        QT_NO_CAST_FROM_ASCII
    PUBLIC_LIBRARIES
        Qt::CorePrivate
        Qt::DeviceDiscoverySupportPrivate
        Qt::GuiPrivate
    PRECOMPILED_HEADER
        "../../corelib/global/qt_pch.h"
)
